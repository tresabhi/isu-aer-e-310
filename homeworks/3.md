$$

$$

$$

$$
u = \frac{\partial \psi}{\partial y}
$$

$$
\partial \psi = u \partial y = 6xy \partial y
$$

$$
\psi = 3xy^2 + f(x)
$$

$$
v = -\frac{\partial \psi}{\partial x} = -\frac{\partial (3xy^2 + f(x))}{\partial x}
$$

$$
v = \cancel{3y^2} + f'(x) = 2x - \cancel{3y^3}
$$

$$
f'(x) = 2x
$$

$$
f(x) = x^2 + c
$$

$$
\boxed{\psi = 3xy^2 + x^2 + c}
